```python
"""原価計算プロトタイプ — Streamlit メインアプリ (ダッシュボード中心)."""
from __future__ import annotations

import os
import sys
from pathlib import Path

sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

import streamlit as st

st.set_page_config(
    page_title="原価計算プロトタイプ",
    page_icon="🧮",
    layout="wide",
    initial_sidebar_state="expanded",
)

from proto.storage.scenario_repo import ScenarioRepository
from proto.ui.sidebar import render_sidebar
from proto.ui.dashboard import render_dashboard
from proto.ui.section_basic import render_basic_info
from proto.ui.section_items import render_items
from proto.ui.section_gift import render_gifts
from proto.ui.section_result import render_results
from proto.ui.scenarios import render_scenarios
from proto.ui.section_verify import render_verify


@st.cache_resource
def get_scenario_repo() -> ScenarioRepository:
    default_path = Path(__file__).parent / "data" / "scenarios.db"
    db_path = os.environ.get("TAMATEX_SCENARIO_DB", str(default_path))
    return ScenarioRepository(db_path)


def main() -> None:
    st.title("輸入原価計算プロトタイプ")
    st.caption("ダッシュボード型 UI — 意思決定支援版")

    repo = get_scenario_repo()
    cond = render_sidebar(repo=repo)

    # 計算は従来通り、タブ描画前に実行
    from proto.data.mock_items import MOCK_ITEMS
    from proto.data.mock_gifts import MOCK_GIFTS
    from proto.engine.calc_single import calc_single_cost
    from proto.engine.calc_gift import calc_gift_cost
    from proto.engine.calc_summary import calc_summary

    items = MOCK_ITEMS
    item_results = [calc_single_cost(it, cond) for it in items]
    gifts = MOCK_GIFTS
    gift_results = [calc_gift_cost(g, cond, item_results) for g in gifts]
    summary = calc_summary(gifts, gift_results)

    tabs = st.tabs([
        "🎯 ダッシュボード",
        "👕 単品タオル",
        "🎁 ギフトセット",
        "📊 比較一覧",
        "💾 シナリオ",
        "✅ Excel検証",
    ])

    with tabs[0]:
        render_dashboard(cond, items, item_results, gifts, gift_results, summary)
    with tabs[1]:
        render_items(cond)  # 既存シグネチャ維持
    with tabs[2]:
        render_gifts(cond, items, item_results)
    with tabs[3]:
        render_results(gifts, gift_results)
    with tabs[4]:
        render_scenarios(cond, items, item_results, gifts, gift_results, repo)
    with tabs[5]:
        render_verify(item_results, gift_results)


if __name__ == "__main__":
    main()
```

※ 既存の `render_items` / `render_gifts` 等のシグネチャが異なる場合は、既存挙動を壊さないように引数を合わせること。

```python
"""Smoke test — UI モジュールと ScenarioRepository の基本動作."""
from __future__ import annotations

from pathlib import Path

import pytest


def test_all_ui_modules_import() -> None:
    from proto.ui import (
        dashboard,
        scenarios,
        sidebar,
        section_basic,
        section_items,
        section_gift,
        section_result,
        section_verify,
    )
    from proto.ui.components import kpi_cards, aggrid_table, sensitivity_chart, styles
    # import に成功すれば OK
    assert dashboard.render_dashboard is not None
    assert scenarios.render_scenarios is not None


def test_scenario_repo_crud(tmp_path: Path) -> None:
    from proto.data.mock_params import COND_20FT
    from proto.storage.scenario_repo import ScenarioRepository

    repo = ScenarioRepository(tmp_path / "smoke.db")
    try:
        sid = repo.save_scenario("smoke_test", COND_20FT)
        s = repo.get_scenario(sid)
        assert s.name == "smoke_test"
        assert s.condition == COND_20FT
        repo.delete_scenario(sid)
        assert len(repo.list_scenarios()) == 0
    finally:
        repo.close()
```
